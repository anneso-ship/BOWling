
	
	public Text Timing;
	public float time = 5f;
	//bool CountDownOn = true;

	
	void Start(){
		StartCoroutine(timer());
		time+=1;
	}

	IEnumerator timer(){
		while(time>0){
			time--;
			yield return new WaitForSeconds(1f);
			Timing.text = ""+time;	
		}
	}

	Update(){//METTRE CA DANS LE ONCOLLISIONENTER
		if(time==0){
			resetBallPosition();
		}
	}
